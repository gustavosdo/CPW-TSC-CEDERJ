{-----------------------------------------------------------------}
{               executa a busca dos transistores                  }
{-----------------------------------------------------------------}

function buscatran (ntr : integer) : PLTRAN;

var
   pltr  : PLTRAN;
   achei : boolean;

begin
   achei := false;
   pltr := listrans;
   while (pltr <> nil) and (not achei) do
      if pltr^.trans.numtran = ntr then
         achei := true
      else
         pltr := pltr^.trprox;

   buscatran := pltr;
end;
